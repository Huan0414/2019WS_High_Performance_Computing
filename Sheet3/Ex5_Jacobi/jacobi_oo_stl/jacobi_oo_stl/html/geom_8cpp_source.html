142  int const kx = _myid % _procy; // MOD(myid,procx)
143  // Determine the neighbors of domain/rank myid

145  _neigh[1] = (kx == _procx - 1) ? -1 : _myid + 1; // East

153  double const hy = 1. / _procy;

155  _xr = (kx + 1) * hx; // right
156  _yb = ky * hy; // bottom
